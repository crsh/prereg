% Subsubsection: inline bold heading so it shares a line with the leftmargin
% label (via titlesec's @noskipsec mechanism).  A forced \par after the title
% text starts body content on the next line.  When the heading appears between
% paragraphs (v-mode) a small vertical skip is added first.
\makeatletter
\renewcommand{\subsubsection}{\@ifstar\@prereg@sss\@prereg@sss}
\newcommand{\@prereg@sss}[1]{%
  \ifhmode
    \textbf{#1}\par\ignorespaces
  \else
    \addvspace{1.5ex plus 0.5ex minus 0.3ex}\noindent\textbf{#1}\par\ignorespaces
  \fi}
\makeatother
